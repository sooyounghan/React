### 카운터 앱 - UI 구현하기
-----
1. Viewer.jsx
```jsx
const Viewer = () => {
  return (
    <div>
      <div>현재 카운트 : </div>
      <h1> 0 </h1>
    </div>
  );
};

export default Viewer;
```

2. Controller.jsx
```jsx
const Controller = () => {
  return (
    <div>
      <button> -1 </button>
      <button> -10 </button>
      <button> -100 </button>
      <button> +100 </button>
      <button> +10 </button>
      <button> +1 </button>
    </div>
  );
};

export default Controller;
```

3. App.jsx
```jsx
import "./App.css";
import Viewer from "./components/Viewer";
import Controller from "./components/Controller";

function App() {
  return (
    <div className="App">
      <h1> Simple Counter </h1>
      <section>
        <Viewer />
      </section>
      <section>
        <Controller />
      </section>
    </div>
  );
}

export default App;
```

4. App.css
```css
body {
  padding: 20px;
}

.App {
  margin: 0 auto; /* 설정한 너비 외 양족으로 남는 부분은 좌우로 margin을 auto로 설정하여 가운데 정렬*/
  width: 400px; /* 너비를 400px로 설정 */
}
.App > section {
  background-color: rgb(245, 245, 245);
  border: 1px solid rgb(240, 240, 240);
  border-radius: 5px;
  padding: 20px;
  margin-bottom: 10px;
}
```
